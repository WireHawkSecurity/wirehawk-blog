Without this, in-page anchor jumps (e.g. the ToC "Final Thoughts" link) land short because images above the target load in and push it down, and the scroll-spy measures a moving layout. Raster images only; SVGs have no processable Width/Height. */ -}} {{- if and (eq .ResourceType "image") (not (in .MediaType.SubType "svg")) -}} {{- $dims = dict "width" (string .Width) "height" (string .Height) -}} {{- end -}} {{- end -}} {{- end -}} {{- $attributes := merge .Attributes (dict "alt" .Text "src" $src "title" (.Title | transform.HTMLEscape) "loading" "lazy") $dims -}} {{- /**/ -}}
